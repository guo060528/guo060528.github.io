function previewImage(img) {
    var fullImage = document.querySelector('.full-image');
    fullImage.querySelector('img').src = img.src;
    fullImage.classList.add('show');
}

function closeImagePreview() {
    var fullImage = document.querySelector('.full-image');
    fullImage.classList.remove('show');
}
